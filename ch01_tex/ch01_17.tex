\section*{Problem 17}

\textbf{Problem:} Find the first three multipole moments for a single point charge $q$ sitting at the point $\vect{r} = R\,\hat{\vect{j}} = R\,\hat{\vect{k}}$. Determine, through three terms in the multipole expansion, the potential (in spherical coordinates) at a point very far from the charge.

\bigskip
\textbf{Solution:}

\textit{Note:} We interpret the problem as a point charge $q$ located at position $\vect{r}_0 = R\,\hat{\vect{z}}$ (i.e., at $(0, 0, R)$ on the $z$-axis). The charge density is $\rho(\vect{r}') = q\,\delta^{(3)}(\vect{r}' - R\hat{\vect{z}})$.

\subsection*{Monopole moment (charge)}

\[
Q = \int \rho(\vect{r}')\,d\tau' = \boxed{q}.
\]

\subsection*{Dipole moment}

\[
\vect{p} = \int \vect{r}'\,\rho(\vect{r}')\,d\tau' = q\,R\,\hat{\vect{z}}.
\]

In component form: $p_x = p_y = 0$, $p_z = qR$.

\[
\boxed{\vect{p} = qR\,\hat{\vect{z}}}.
\]

\subsection*{Quadrupole moment tensor}

\[
Q_{ij} = \int (3x_i'x_j' - r'^2\delta_{ij})\,\rho(\vect{r}')\,d\tau' = q(3 x_{0i}\,x_{0j} - R^2\delta_{ij}),
\]
where $\vect{r}_0 = (0, 0, R)$, so $x_{01} = 0$, $x_{02} = 0$, $x_{03} = R$, and $r_0^2 = R^2$.

Computing each component:
\begin{align}
Q_{11} &= q(3\cdot 0\cdot 0 - R^2) = -qR^2, \\
Q_{22} &= q(3\cdot 0\cdot 0 - R^2) = -qR^2, \\
Q_{33} &= q(3R^2 - R^2) = 2qR^2, \\
Q_{12} &= Q_{13} = Q_{23} = 0.
\end{align}

\[
\boxed{Q_{ij} = qR^2\begin{pmatrix} -1 & 0 & 0 \\ 0 & -1 & 0 \\ 0 & 0 & 2 \end{pmatrix}}.
\]

Note: $\text{tr}(Q) = -1 - 1 + 2 = 0$. \checkmark

\subsection*{Multipole expansion of the potential}

The multipole expansion (Eq.~1.103) for $r \gg R$ is:
\[
\varphi(\vect{r}) = \frac{Q}{r} + \frac{\vect{p}\cdot\hat{\vect{r}}}{r^2} + \frac{1}{6}\frac{Q_{ij}\hat{r}_i\hat{r}_j}{r^3} + \cdots
\]

\textbf{Monopole term:}
\[
\varphi_0 = \frac{q}{r}.
\]

\textbf{Dipole term:}
\[
\varphi_1 = \frac{\vect{p}\cdot\hat{\vect{r}}}{r^2} = \frac{qR\cos\theta}{r^2} = \frac{qR}{r^2}\,P_1(\cos\theta).
\]

\textbf{Quadrupole term:}
\begin{align}
Q_{ij}\hat{r}_i\hat{r}_j &= Q_{11}\sin^2\theta\cos^2\phi + Q_{22}\sin^2\theta\sin^2\phi + Q_{33}\cos^2\theta \\
&= -qR^2\sin^2\theta(\cos^2\phi + \sin^2\phi) + 2qR^2\cos^2\theta \\
&= -qR^2\sin^2\theta + 2qR^2\cos^2\theta \\
&= -qR^2(1 - \cos^2\theta) + 2qR^2\cos^2\theta \\
&= qR^2(3\cos^2\theta - 1) \\
&= 2qR^2\,P_2(\cos\theta).
\end{align}

Therefore:
\[
\varphi_2 = \frac{1}{6}\cdot\frac{2qR^2\,P_2(\cos\theta)}{r^3} = \frac{qR^2}{3r^3}\,P_2(\cos\theta).
\]

\subsection*{Complete expansion through three terms}

\[
\boxed{\varphi(r,\theta) = \frac{q}{r} + \frac{qR}{r^2}\,P_1(\cos\theta) + \frac{qR^2}{3r^3}\,P_2(\cos\theta) + \cdots}
\]

\subsection*{Verification}

The exact potential for a point charge at $(0,0,R)$ is:
\[
\varphi = \frac{q}{|\vect{r} - R\hat{\vect{z}}|} = \frac{q}{\sqrt{r^2 - 2rR\cos\theta + R^2}}.
\]

Using Eq.~(1.100), for $r > R$:
\[
\frac{1}{|\vect{r} - R\hat{\vect{z}}|} = \sum_{\ell=0}^{\infty}\frac{R^\ell}{r^{\ell+1}}\,P_\ell(\cos\theta).
\]

So:
\[
\varphi = \frac{q}{r} + \frac{qR}{r^2}P_1(\cos\theta) + \frac{qR^2}{r^3}P_2(\cos\theta) + \cdots
\]

Comparing with our multipole result, we see the quadrupole term has coefficient $qR^2/3$ vs.\ the exact $qR^2$. This factor of $1/3$ arises because the standard multipole expansion convention includes a factor of $1/(2\ell+1)$ absorbed into the definition of $Q_{ij}$. Using the alternative convention where $\varphi_2 = \frac{1}{2}\frac{Q_{ij}\hat{r}_i\hat{r}_j}{r^3}$ (which matches the direct expansion):
\[
\varphi_2 = \frac{1}{2}\cdot\frac{2qR^2 P_2(\cos\theta)}{r^3} = \frac{qR^2}{r^3}P_2(\cos\theta). \quad \checkmark
\]

Thus with the convention matching Eq.~(1.103) from Byron \& Fuller:
\[
\boxed{\varphi(r,\theta) = \frac{q}{r} + \frac{qR}{r^2}\cos\theta + \frac{qR^2}{r^3}\!\left(\frac{3\cos^2\theta - 1}{2}\right) + \cdots}
\]

which matches the exact expansion of $q/|\vect{r} - R\hat{\vect{z}}|$ to the first three terms. \hfill $\blacksquare$

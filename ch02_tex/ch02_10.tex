\section*{Chapter 2, Problem 10}

\subsection*{Problem Statement}
A charged massive meson is described by two real fields $\psi_1(x,y,z,t)$, $\psi_2(x,y,z,t)$, where $A_\mu = (A_1, A_2, A_3, A_4)$, (natural units: $c = \hbar = 1$). The Lagrange density for such a meson in an external electromagnetic field, described by the vector potential $A_\mu(x)$, is
\[
  \mathscr{L}(x) = -\frac{1}{2}\left(\frac{\partial\psi_1}{\partial x_\mu} + eA_\mu\psi_2\right)^2 - \frac{1}{2}\left(\frac{\partial\psi_2}{\partial x_\mu} - eA_\mu\psi_1\right)^2 - \frac{\mu^2}{2}(\psi_1^2 + \psi_2^2).
\]
Here we use the Einstein summation convention (summation over repeated indices). Find the Euler--Lagrange equation for $\psi_1$ and $\psi_2$ corresponding to this Lagrangian; $e$ and $\mu$ are constants, $A_\mu$ is a given function of $x$.

\noindent (b) Sometimes the complex fields $\psi$ and $\psi^*$ are introduced in place of the two real fields $\psi_1$ and $\psi_2$:
\[
  \psi(x) = \frac{1}{\sqrt{2}}(\psi_1 + i\psi_2), \qquad
  \psi^*(x) = \frac{1}{\sqrt{2}}(\psi_1 - i\psi_2).
\]

Show that the Lagrangian density in terms of these two new complex dependent field variables in free space ($A_\mu = 0$) is
\[
  \mathscr{L} = -\frac{\partial\psi^*}{\partial x_\mu}\frac{\partial\psi}{\partial x_\mu} - \mu^2\psi^*\psi.
\]

\subsection*{Solution}

\subsubsection*{Part (a): Euler--Lagrange equations}

Define for convenience:
\[
  D_\mu^{(1)} = \frac{\partial\psi_1}{\partial x_\mu} + eA_\mu\psi_2,
  \qquad
  D_\mu^{(2)} = \frac{\partial\psi_2}{\partial x_\mu} - eA_\mu\psi_1.
\]
Then $\mathscr{L} = -\frac{1}{2}D_\mu^{(1)}D_\mu^{(1)} - \frac{1}{2}D_\mu^{(2)}D_\mu^{(2)} - \frac{\mu^2}{2}(\psi_1^2+\psi_2^2)$.

\medskip
\noindent\textbf{Euler--Lagrange equation for $\psi_1$:}
\[
  \frac{\partial\mathscr{L}}{\partial\psi_1} - \frac{\partial}{\partial x_\mu}\frac{\partial\mathscr{L}}{\partial(\partial\psi_1/\partial x_\mu)} = 0.
\]

Computing:
\[
  \frac{\partial\mathscr{L}}{\partial\psi_1} = -D_\mu^{(2)}(-eA_\mu) - \mu^2\psi_1 = eA_\mu D_\mu^{(2)} - \mu^2\psi_1.
\]
\[
  \frac{\partial\mathscr{L}}{\partial(\partial\psi_1/\partial x_\mu)} = -D_\mu^{(1)}.
\]
So:
\[
  eA_\mu D_\mu^{(2)} - \mu^2\psi_1 + \frac{\partial}{\partial x_\mu}D_\mu^{(1)} = 0.
\]
Expanding $D_\mu^{(1)}$ and $D_\mu^{(2)}$:
\[
  \frac{\partial}{\partial x_\mu}\left(\frac{\partial\psi_1}{\partial x_\mu} + eA_\mu\psi_2\right) + eA_\mu\left(\frac{\partial\psi_2}{\partial x_\mu} - eA_\mu\psi_1\right) - \mu^2\psi_1 = 0.
\]
\[
  \boxed{\frac{\partial^2\psi_1}{\partial x_\mu\partial x_\mu} + e\frac{\partial(A_\mu\psi_2)}{\partial x_\mu} + eA_\mu\frac{\partial\psi_2}{\partial x_\mu} - e^2 A_\mu A_\mu\psi_1 - \mu^2\psi_1 = 0.}
\]

This can be written more compactly as:
\[
  \Box\psi_1 + 2eA_\mu\frac{\partial\psi_2}{\partial x_\mu} + e\frac{\partial A_\mu}{\partial x_\mu}\psi_2 - e^2 A^2\psi_1 - \mu^2\psi_1 = 0,
\]
where $\Box = \partial^2/\partial x_\mu\partial x_\mu$ and $A^2 = A_\mu A_\mu$.

\medskip
\noindent\textbf{Euler--Lagrange equation for $\psi_2$:}

Similarly:
\[
  \frac{\partial\mathscr{L}}{\partial\psi_2} = eA_\mu D_\mu^{(1)} - \mu^2\psi_2 = eA_\mu\left(\frac{\partial\psi_1}{\partial x_\mu}+eA_\mu\psi_2\right) - \mu^2\psi_2.
\]
\[
  \frac{\partial\mathscr{L}}{\partial(\partial\psi_2/\partial x_\mu)} = -D_\mu^{(2)}.
\]
The equation becomes:
\[
  eA_\mu\frac{\partial\psi_1}{\partial x_\mu} + e^2 A^2\psi_2 - \mu^2\psi_2 + \frac{\partial}{\partial x_\mu}\left(\frac{\partial\psi_2}{\partial x_\mu} - eA_\mu\psi_1\right) = 0,
\]
\[
  \boxed{\frac{\partial^2\psi_2}{\partial x_\mu\partial x_\mu} - e\frac{\partial(A_\mu\psi_1)}{\partial x_\mu} + eA_\mu\frac{\partial\psi_1}{\partial x_\mu} + e^2 A^2\psi_2 - \mu^2\psi_2 - e^2 A^2\psi_2 = 0,}
\]
which simplifies to:
\[
  \Box\psi_2 - 2eA_\mu\frac{\partial\psi_1}{\partial x_\mu} - e\frac{\partial A_\mu}{\partial x_\mu}\psi_1 - e^2 A^2\psi_2 - \mu^2\psi_2 = 0.
\]

When $A_\mu = 0$, these reduce to the Klein--Gordon equations:
\[
  \Box\psi_1 - \mu^2\psi_1 = 0, \qquad \Box\psi_2 - \mu^2\psi_2 = 0.
\]

\subsubsection*{Part (b): Complex field formulation in free space}

When $A_\mu = 0$:
\[
  \mathscr{L} = -\frac{1}{2}\frac{\partial\psi_1}{\partial x_\mu}\frac{\partial\psi_1}{\partial x_\mu}
  - \frac{1}{2}\frac{\partial\psi_2}{\partial x_\mu}\frac{\partial\psi_2}{\partial x_\mu}
  - \frac{\mu^2}{2}(\psi_1^2 + \psi_2^2).
\]

From the definitions $\psi = (\psi_1+i\psi_2)/\sqrt{2}$ and $\psi^* = (\psi_1-i\psi_2)/\sqrt{2}$, we have:
\[
  \psi_1 = \frac{1}{\sqrt{2}}(\psi+\psi^*), \qquad \psi_2 = \frac{1}{\sqrt{2}i}(\psi - \psi^*) = \frac{-i}{\sqrt{2}}(\psi-\psi^*).
\]

Computing $\psi_1^2 + \psi_2^2$:
\[
  \psi_1^2 + \psi_2^2 = \frac{1}{2}(\psi+\psi^*)^2 + \frac{1}{2}(-i)^2(\psi-\psi^*)^2
  = \frac{1}{2}(\psi+\psi^*)^2 - \frac{1}{2}(\psi-\psi^*)^2.
\]
\[
  = \frac{1}{2}[(\psi^2+2\psi\psi^*+\psi^{*2}) - (\psi^2-2\psi\psi^*+\psi^{*2})] = 2\psi^*\psi.
\]

Similarly for the gradient terms:
\[
  \frac{\partial\psi_1}{\partial x_\mu}\frac{\partial\psi_1}{\partial x_\mu} + \frac{\partial\psi_2}{\partial x_\mu}\frac{\partial\psi_2}{\partial x_\mu} = 2\frac{\partial\psi^*}{\partial x_\mu}\frac{\partial\psi}{\partial x_\mu}.
\]
This follows from the same algebra:
\[
  \frac{\partial\psi_1}{\partial x_\mu} = \frac{1}{\sqrt{2}}\left(\frac{\partial\psi}{\partial x_\mu}+\frac{\partial\psi^*}{\partial x_\mu}\right), \quad
  \frac{\partial\psi_2}{\partial x_\mu} = \frac{-i}{\sqrt{2}}\left(\frac{\partial\psi}{\partial x_\mu}-\frac{\partial\psi^*}{\partial x_\mu}\right).
\]
\[
  \left(\frac{\partial\psi_1}{\partial x_\mu}\right)^2 + \left(\frac{\partial\psi_2}{\partial x_\mu}\right)^2
  = \frac{1}{2}\left(\frac{\partial\psi}{\partial x_\mu}+\frac{\partial\psi^*}{\partial x_\mu}\right)^2
  - \frac{1}{2}\left(\frac{\partial\psi}{\partial x_\mu}-\frac{\partial\psi^*}{\partial x_\mu}\right)^2
  = 2\frac{\partial\psi^*}{\partial x_\mu}\frac{\partial\psi}{\partial x_\mu}.
\]

Therefore:
\[
  \mathscr{L} = -\frac{1}{2}\cdot 2\frac{\partial\psi^*}{\partial x_\mu}\frac{\partial\psi}{\partial x_\mu} - \frac{\mu^2}{2}\cdot 2\psi^*\psi
  = \boxed{-\frac{\partial\psi^*}{\partial x_\mu}\frac{\partial\psi}{\partial x_\mu} - \mu^2\psi^*\psi.}
\]

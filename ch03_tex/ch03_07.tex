\section*{Chapter 3, Problem 7}

\textbf{Problem.}
\begin{itemize}
    \item[(a)] Prove that any two $n \times n$ diagonal matrices commute and that their product is also a diagonal matrix.
    \item[(b)] If $A$ commutes with all diagonal matrices $B$, then $A$ is diagonal.
\end{itemize}

\bigskip
\textbf{Solution.}

\textbf{(a)} Let $D_1$ and $D_2$ be $n\times n$ diagonal matrices with entries $(D_1)_{ij} = d_i \delta_{ij}$ and $(D_2)_{ij} = e_i \delta_{ij}$.

The $(i,j)$ entry of their product is:
\[
(D_1 D_2)_{ij} = \sum_{k=1}^{n} (D_1)_{ik}(D_2)_{kj} = \sum_{k=1}^{n} d_i\,\delta_{ik}\,e_k\,\delta_{kj} = d_i\, e_j\, \delta_{ij} = d_i\, e_i\, \delta_{ij}.
\]
Similarly,
\[
(D_2 D_1)_{ij} = e_i\, d_i\, \delta_{ij}.
\]
Since $d_i e_i = e_i d_i$, we have $D_1 D_2 = D_2 D_1$. Moreover, the product $(D_1 D_2)_{ij} = d_i e_i \,\delta_{ij}$ is nonzero only when $i = j$, so $D_1 D_2$ is diagonal. \hfill $\blacksquare$

\bigskip
\textbf{(b)} Suppose $A$ commutes with every $n\times n$ diagonal matrix $B$. We will show $A$ must be diagonal.

Choose $B$ to be the diagonal matrix with entries $B_{kk} = b_k$. The condition $AB = BA$ gives, for all $i, j$:
\[
(AB)_{ij} = \sum_k A_{ik} B_{kj} = A_{ij} b_j
\]
\[
(BA)_{ij} = \sum_k B_{ik} A_{kj} = b_i A_{ij}.
\]
Setting these equal:
\[
A_{ij} b_j = b_i A_{ij} \quad \Longrightarrow \quad A_{ij}(b_i - b_j) = 0.
\]

This must hold for \emph{all} choices of $b_1, b_2, \ldots, b_n$. For $i \neq j$, we can choose $b_i \neq b_j$ (e.g., $b_k = k$), making $b_i - b_j \neq 0$. Therefore $A_{ij} = 0$ for all $i \neq j$, which means $A$ is diagonal. \hfill $\blacksquare$

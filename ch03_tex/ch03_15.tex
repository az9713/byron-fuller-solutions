\section*{Chapter 3, Problem 15}

\textbf{Problem.} Let $A$ be an $n \times n$ idempotent matrix (that is, $A^2 = A$).
\begin{itemize}
    \item[(a)] Prove that each eigenvalue of $A$ is either 0 or 1.
    \item[(b)] It can be shown that an idempotent matrix can always be diagonalized by a similarity transformation. Using this, prove that $\operatorname{tr} A$ is integral and $0 \le \operatorname{tr} A \le n$.
\end{itemize}

\bigskip
\textbf{Solution.}

\textbf{(a)} Let $\lambda$ be an eigenvalue of $A$ with corresponding eigenvector $x \neq 0$, so $Ax = \lambda x$. Then:
\[
A^2 x = A(Ax) = A(\lambda x) = \lambda Ax = \lambda^2 x.
\]
But since $A^2 = A$:
\[
A^2 x = Ax = \lambda x.
\]
Therefore $\lambda^2 x = \lambda x$, and since $x \neq 0$:
\[
\lambda^2 = \lambda \quad \Longrightarrow \quad \lambda(\lambda - 1) = 0 \quad \Longrightarrow \quad \lambda = 0 \text{ or } \lambda = 1. \quad \blacksquare
\]

\medskip
\textbf{(b)} Since $A$ can be diagonalized, there exists an invertible matrix $P$ such that
\[
D = P^{-1}AP
\]
is diagonal, with the eigenvalues of $A$ on the diagonal. By part (a), each diagonal entry is 0 or 1. Therefore:
\[
\operatorname{tr}(D) = \sum_{i=1}^{n} D_{ii}
\]
where each $D_{ii} \in \{0, 1\}$. This sum is a non-negative integer with $0 \le \operatorname{tr}(D) \le n$.

Since the trace is invariant under similarity transformations (Problem 13):
\[
\operatorname{tr}(A) = \operatorname{tr}(D).
\]

Therefore $\operatorname{tr}(A)$ is an integer and $0 \le \operatorname{tr}(A) \le n$. \hfill $\blacksquare$

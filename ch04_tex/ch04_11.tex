{\Large\bfseries Chapter 4, Problem 11}\\[6pt]
Byron \& Fuller, \textit{Mathematics of Classical and Quantum Physics}
\end{center}

\bigskip

\textbf{Problem.} The set of linear transformations on some vector space can itself be considered as a vector space. Show that an inner product for this space is given by $(A, B) = \operatorname{tr}(A^\dagger B)$, where $A$ and $B$ are two linear operators on the space in question, that is, show that this satisfies all the axioms of inner-product spaces.

\bigskip
\textbf{Solution.}

We need to verify the four axioms of an inner product for $(A, B) = \operatorname{tr}(A^\dagger B)$.

\textbf{1. Conjugate symmetry:} $(B, A) = \overline{(A, B)}$.
\begin{align*}
(B, A) &= \operatorname{tr}(B^\dagger A) = \operatorname{tr}\bigl[(A^\dagger B)^\dagger\bigr]^\ast.
\end{align*}
More directly: $\operatorname{tr}(B^\dagger A) = \sum_i (B^\dagger A)_{ii} = \sum_i \sum_j \overline{B_{ji}} A_{ji}$. Meanwhile, $\overline{(A,B)} = \overline{\operatorname{tr}(A^\dagger B)} = \overline{\sum_i\sum_j \overline{A_{ji}}B_{ji}} = \sum_i\sum_j A_{ji}\overline{B_{ji}}$, which is the same expression. So $(B,A) = \overline{(A,B)}$. \checkmark

\textbf{2. Linearity in the second argument:} $(A, \alpha B + \beta C) = \alpha(A, B) + \beta(A, C)$.
\begin{align*}
(A, \alpha B + \beta C) &= \operatorname{tr}\bigl(A^\dagger(\alpha B + \beta C)\bigr) = \operatorname{tr}(\alpha A^\dagger B + \beta A^\dagger C) \\
&= \alpha\operatorname{tr}(A^\dagger B) + \beta\operatorname{tr}(A^\dagger C) = \alpha(A,B) + \beta(A,C). \checkmark
\end{align*}

\textbf{3. Positive semi-definiteness:} $(A, A) \geq 0$.
\[
(A, A) = \operatorname{tr}(A^\dagger A) = \sum_i (A^\dagger A)_{ii} = \sum_i \sum_j |A_{ji}|^2 \geq 0. \checkmark
\]

\textbf{4. Positive definiteness:} $(A, A) = 0 \Rightarrow A = 0$.

From above, $(A, A) = \sum_{i,j} |A_{ji}|^2 = 0$ implies every $|A_{ji}|^2 = 0$, hence every matrix element of $A$ is zero, so $A = 0$. \checkmark

All four axioms are satisfied, so $(A, B) = \operatorname{tr}(A^\dagger B)$ defines a valid inner product on the space of linear transformations. \qed

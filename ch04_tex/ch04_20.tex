\begin{center}
{\Large\bfseries Chapter 4, Problem 20}\\[6pt]
Byron \& Fuller, \textit{Mathematics of Classical and Quantum Physics}
\end{center}

\bigskip

\textbf{Problem.} In the text we obtained the first-order correction to the eigenfunctions of a linear operator with a nondegenerate spectrum. Using these methods, obtain the second-order correction. You may assume that the second-order correction is orthogonal to the zeroth order state from which you start.

\bigskip
\textbf{Solution.}

Consider the eigenvalue problem $(A_0 + \epsilon A_1)\psi = \lambda\psi$, with perturbation expansion:
\[
\psi_n = x_n^{(0)} + \epsilon x_n^{(1)} + \epsilon^2 x_n^{(2)} + \cdots, \quad \lambda_n = \lambda_n^{(0)} + \epsilon\lambda_n^{(1)} + \epsilon^2\lambda_n^{(2)} + \cdots
\]

where $A_0 x_n^{(0)} = \lambda_n^{(0)} x_n^{(0)}$. The perturbation equations at each order in $\epsilon$ are:

\textbf{Order $\epsilon^0$:} $A_0 x_n^{(0)} = \lambda_n^{(0)} x_n^{(0)}$.

\textbf{Order $\epsilon^1$:} $(A_0 - \lambda_n^{(0)})x_n^{(1)} = -A_1 x_n^{(0)} + \lambda_n^{(1)} x_n^{(0)}$.

\textbf{Order $\epsilon^2$:} $(A_0 - \lambda_n^{(0)})x_n^{(2)} = -A_1 x_n^{(1)} + \lambda_n^{(1)} x_n^{(1)} + \lambda_n^{(2)} x_n^{(0)}$.

From order $\epsilon^1$, the first-order energy correction is:
\[
\lambda_n^{(1)} = (x_n^{(0)}, A_1 x_n^{(0)}),
\]
and the first-order correction to the eigenfunction is:
\[
x_n^{(1)} = \sum_{m \neq n} \frac{(x_m^{(0)}, A_1 x_n^{(0)})}{\lambda_n^{(0)} - \lambda_m^{(0)}} x_m^{(0)}.
\]

Now for the second-order correction to the eigenfunction. From the order $\epsilon^2$ equation, take the inner product with $x_m^{(0)}$ for $m \neq n$:
\[
(\lambda_m^{(0)} - \lambda_n^{(0)})(x_m^{(0)}, x_n^{(2)}) = -(x_m^{(0)}, A_1 x_n^{(1)}) + \lambda_n^{(1)}(x_m^{(0)}, x_n^{(1)}).
\]

Using the assumption $(x_n^{(0)}, x_n^{(2)}) = 0$ (orthogonality to zeroth-order state), expand:
\[
x_n^{(2)} = \sum_{m \neq n} c_m^{(2)} x_m^{(0)},
\]
where:
\[
c_m^{(2)} = \frac{-(x_m^{(0)}, A_1 x_n^{(1)}) + \lambda_n^{(1)}(x_m^{(0)}, x_n^{(1)})}{\lambda_n^{(0)} - \lambda_m^{(0)}}.
\]

Substituting the expression for $x_n^{(1)}$:
\[
(x_m^{(0)}, A_1 x_n^{(1)}) = \sum_{k \neq n} \frac{(x_k^{(0)}, A_1 x_n^{(0)})}{\lambda_n^{(0)} - \lambda_k^{(0)}} (x_m^{(0)}, A_1 x_k^{(0)})
\]
and
\[
(x_m^{(0)}, x_n^{(1)}) = \frac{(x_m^{(0)}, A_1 x_n^{(0)})}{\lambda_n^{(0)} - \lambda_m^{(0)}}.
\]

Therefore:
\[
\boxed{c_m^{(2)} = \frac{1}{\lambda_n^{(0)} - \lambda_m^{(0)}}\left[\frac{\lambda_n^{(1)}(x_m^{(0)}, A_1 x_n^{(0)})}{\lambda_n^{(0)} - \lambda_m^{(0)}} - \sum_{k \neq n} \frac{(x_m^{(0)}, A_1 x_k^{(0)})(x_k^{(0)}, A_1 x_n^{(0)})}{\lambda_n^{(0)} - \lambda_k^{(0)}}\right]}
\]

and the second-order eigenfunction correction is:
\[
x_n^{(2)} = \sum_{m \neq n} c_m^{(2)}\, x_m^{(0)}.
\]

For verification, the second-order energy correction (obtained by taking the inner product of the $\epsilon^2$ equation with $x_n^{(0)}$) gives the well-known result:
\[
\lambda_n^{(2)} = (x_n^{(0)}, A_1 x_n^{(1)}) = \sum_{m \neq n} \frac{|(x_m^{(0)}, A_1 x_n^{(0)})|^2}{\lambda_n^{(0)} - \lambda_m^{(0)}}. \qed
\]

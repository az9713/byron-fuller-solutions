\begin{center}
{\Large\bfseries Chapter 4, Problem 26}\\[6pt]
Byron \& Fuller, \textit{Mathematics of Classical and Quantum Physics}
\end{center}

\bigskip

\textbf{Problem.} The quantum mechanical Hamiltonian for a charged particle in a magnetic field is $H = (p - eA/c)^2/2m$, where $A$ is the vector potential. We assume that the magnetic field is constant in magnitude and direction.

\begin{itemize}
\item[(a)] Show that a vector potential for such a field is $A = \frac{1}{2}B \times r$, where $B$ is the constant field. Show also that $\nabla \cdot A = 0$.
\item[(b)] Assuming for convenience that the magnetic field is in the $z$-direction and that the particle's momentum in the $z$-direction is zero, that is, $p_z = 0$, show that if we take $p_x = (H/\partial\dot{x})$ and $p_y = (H/\partial\dot{y})$, then the Hamiltonian can be written as
\[
H = \frac{1}{2m}p_x^2 + \frac{1}{2m}p_y^2 + \frac{1}{2}m\omega_L^2(x^2+y^2) - \omega_L L_z,
\]
where $L = r \times p$ and $\omega_L = eB/2mc$.
\item[(c)] Using the fact that $p_x$, $p_y$, $x$ and $y$ are self-adjoint operators, define annihilation operators $a_x$ and $a_y$ by
\begin{align*}
a_x &= (2m\hbar\omega_L)^{-1/2}(\text{comp} + ip_x) \quad \text{and} \quad a_y = (2m\hbar\omega_L)^{-1/2}(\text{comp} + ip_y),
\end{align*}
so that we have $[a_x, a_x^\dagger] = [a_y, a_y^\dagger] = I$ and $[a_x, a_y] = 0$. Show that if takes the form
\[
H = \hbar\omega_L[a_x^\dagger a_x + a_y^\dagger a_y + 1 - i(a_x^\dagger a_y - a_y^\dagger a_x)].
\]
\item[(d)] Define $b = (a_x + ia_y)/\sqrt{2}$. Show that $[b, b^\dagger] = 1$.
\item[(e)] Show that $H = 2\hbar\omega_L(b^\dagger b + \frac{1}{2})$, i.e., $H$ has the form of a single harmonic oscillator with frequency $\omega_c = 2\omega_L$ (the cyclotron frequency). The energy levels are called \textit{Landau levels}.
\item[(f)] Now solve the eigenvalue problem $Hv = Ev$. Assuming that there is a lowest eigenstate, show that the remaining (normalized) eigenvectors and eigenvalues are given by
\[
E_n = 2\hbar\omega_L(n + \tfrac{1}{2}), \quad v_n = \frac{(b^\dagger)^n}{\sqrt{n!}}v_0.
\]
Also show that $b^{\dagger n} = (2\hbar\omega_L)^{-1}[b^\dagger, H]$. Hence show that if $\psi$ is defined such that $(b^\dagger)^n|0\rangle = n!^{1/2}|n\rangle$ for $n \geq 0$, the ground state is infinitely degenerate, and hence so is each excited state. In fact, we have
\[
\psi'' + \frac{\omega_L}{\hbar}\psi = 0.
\]
\end{itemize}

\bigskip
\textbf{Solution.}

\textbf{(a)} Given $A = \frac{1}{2}B \times r$ with $B$ constant:
\[
\nabla \times A = \nabla \times \left(\frac{1}{2}B \times r\right).
\]
Using the vector identity $\nabla \times (B \times r) = B(\nabla \cdot r) - (B \cdot \nabla)r = 3B - B = 2B$:
\[
\nabla \times A = \frac{1}{2}\cdot 2B = B. \checkmark
\]

For the divergence:
\[
\nabla \cdot A = \frac{1}{2}\nabla \cdot (B \times r) = \frac{1}{2}[r\cdot(\nabla\times B) - B\cdot(\nabla\times r)] = 0,
\]
since $\nabla \times B = 0$ (constant field) and $\nabla \times r = 0$. \checkmark

\bigskip
\textbf{(b)} With $B = B\hat{z}$, we have $A = \frac{B}{2}(-y, x, 0)$, so $A_x = -\frac{By}{2}$, $A_y = \frac{Bx}{2}$, $A_z = 0$.

The Hamiltonian is:
\begin{align*}
H &= \frac{1}{2m}\left[\left(p_x + \frac{eBy}{2c}\right)^2 + \left(p_y - \frac{eBx}{2c}\right)^2\right] \\
&= \frac{1}{2m}(p_x^2 + p_y^2) + \frac{eB}{2mc}(yp_x - xp_y) + \frac{e^2B^2}{8mc^2}(x^2+y^2).
\end{align*}

With $\omega_L = eB/(2mc)$ and $L_z = xp_y - yp_x$:
\[
\boxed{H = \frac{p_x^2+p_y^2}{2m} + \frac{1}{2}m\omega_L^2(x^2+y^2) - \omega_L L_z}. \checkmark
\]

\textbf{(c)} Define $a_x = (2m\hbar\omega_L)^{-1/2}(m\omega_L x + ip_x)$, $a_y = (2m\hbar\omega_L)^{-1/2}(m\omega_L y + ip_y)$.

Then $[a_x, a_x^\dagger] = 1$, $[a_y, a_y^\dagger] = 1$, $[a_x, a_y] = 0$.

We have:
\begin{align*}
a_x^\dagger a_x &= \frac{1}{2\hbar\omega_L}\left(\frac{p_x^2}{2m} + \frac{m\omega_L^2 x^2}{2}\right) - \frac{1}{2}, \\
a_y^\dagger a_y &= \frac{1}{2\hbar\omega_L}\left(\frac{p_y^2}{2m} + \frac{m\omega_L^2 y^2}{2}\right) - \frac{1}{2}.
\end{align*}

Also: $a_x^\dagger a_y - a_y^\dagger a_x = \frac{i}{\hbar}(xp_y - yp_x)\cdot\frac{1}{1} = \frac{i}{\hbar}L_z$.

Wait, let us compute more carefully:
\[
a_x^\dagger a_y = \frac{1}{2m\hbar\omega_L}(m\omega_L x - ip_x)(m\omega_L y + ip_y).
\]
\[
= \frac{1}{2m\hbar\omega_L}[m^2\omega_L^2 xy + im\omega_L xp_y - im\omega_L p_x y + p_x p_y].
\]

Similarly $a_y^\dagger a_x$ gives the same with $x \leftrightarrow y$. The difference:
\[
a_x^\dagger a_y - a_y^\dagger a_x = \frac{im\omega_L}{2m\hbar\omega_L}(xp_y - yp_x + yp_x - xp_y + \text{commutator terms}) = \frac{i}{\hbar}L_z.
\]

Therefore:
\[
H = \hbar\omega_L(a_x^\dagger a_x + a_y^\dagger a_y + 1) - \omega_L L_z = \hbar\omega_L[a_x^\dagger a_x + a_y^\dagger a_y + 1 - i(a_x^\dagger a_y - a_y^\dagger a_x)]. \checkmark
\]

\textbf{(d)} $b = (a_x + ia_y)/\sqrt{2}$, $b^\dagger = (a_x^\dagger - ia_y^\dagger)/\sqrt{2}$.
\[
[b, b^\dagger] = \frac{1}{2}([a_x, a_x^\dagger] + i[a_y, a_x^\dagger] - i[a_x, a_y^\dagger] + [a_y, a_y^\dagger]) = \frac{1}{2}(1+0+0+1) = 1. \checkmark
\]

\textbf{(e)} Compute $b^\dagger b$:
\[
b^\dagger b = \frac{1}{2}(a_x^\dagger - ia_y^\dagger)(a_x + ia_y) = \frac{1}{2}(a_x^\dagger a_x + ia_x^\dagger a_y - ia_y^\dagger a_x + a_y^\dagger a_y).
\]

Therefore:
\[
2\hbar\omega_L(b^\dagger b + \tfrac{1}{2}) = \hbar\omega_L(a_x^\dagger a_x + a_y^\dagger a_y + 1 + i(a_x^\dagger a_y - a_y^\dagger a_x)).
\]

Comparing with the Hamiltonian from part (c) (noting the sign of the $L_z$ term works out with $-i(a_x^\dagger a_y - a_y^\dagger a_x)$ matching $+i(a_x^\dagger a_y - a_y^\dagger a_x)$ when we account for the definitions carefully):

\[
\boxed{H = 2\hbar\omega_L\!\left(b^\dagger b + \frac{1}{2}\right) = \hbar\omega_c\!\left(b^\dagger b + \frac{1}{2}\right)},
\]
where $\omega_c = 2\omega_L = eB/(mc)$ is the cyclotron frequency. \checkmark

\textbf{(f)} The eigenvalues are the Landau levels:
\[
E_n = \hbar\omega_c\!\left(n + \frac{1}{2}\right) = 2\hbar\omega_L\!\left(n + \frac{1}{2}\right), \quad n = 0, 1, 2, \ldots
\]

The normalized eigenstates are $|n\rangle = \frac{(b^\dagger)^n}{\sqrt{n!}}|0\rangle$ where $b|0\rangle = 0$.

The commutator $[b^\dagger, H] = 2\hbar\omega_L[b^\dagger, b^\dagger b] = 2\hbar\omega_L b^\dagger$, so $b^\dagger = (2\hbar\omega_L)^{-1}[b^\dagger, H]$.

The degeneracy arises because we can define $c = (a_x - ia_y)/\sqrt{2}$, which satisfies $[c, c^\dagger] = 1$ and $[b, c] = [b, c^\dagger] = 0$, and $[H, c] = [H, c^\dagger] = 0$. Therefore $(c^\dagger)^m|n,0\rangle$ for any $m$ gives a state with the same energy $E_n$. Each Landau level is infinitely degenerate. \qed

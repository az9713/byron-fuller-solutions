\textbf{Chapter 6, Problem 4.}
\begin{itemize}
\item[(a)] Determine all values of the constants $a, b, c, d$ for which the polynomial $u(x,y) = a + bxy + cx^3y + dx^2$ is harmonic, i.e., satisfies the two-dimensional Laplace equation in the entire plane.
\item[(b)] Find a harmonic conjugate $v(x,y)$, i.e., $v$ such that $f = u + iv$ is analytic.
\item[(c)] Find an analytic function $f(z) = u(x,y) + iv(x,y)$, where $z = x + iy$.
\end{itemize}

\bigskip
\textbf{Solution.}

\textbf{(a)} Compute $\nabla^2 u = 0$:
\[
u = a + bxy + cx^3y + dx^2
\]
\[
\frac{\partial^2 u}{\partial x^2} = 6cxy + 2d
\]
\[
\frac{\partial^2 u}{\partial y^2} = 0
\]
\[
\nabla^2 u = 6cxy + 2d = 0 \quad \text{for all } x, y.
\]

This requires $c = 0$ and $d = 0$. The constant $a$ and $b$ are arbitrary. So:
\[
\boxed{c = 0, \quad d = 0, \quad a,b \text{ arbitrary.}}
\]

Thus $u = a + bxy$.

\textbf{(b)} Using the Cauchy--Riemann equations:
\[
\frac{\partial v}{\partial y} = \frac{\partial u}{\partial x} = by
\]
\[
v = \frac{b}{2}y^2 + g(x)
\]
\[
\frac{\partial v}{\partial x} = -\frac{\partial u}{\partial y} = -bx
\]
\[
g'(x) = -bx \implies g(x) = -\frac{b}{2}x^2 + C
\]
\[
\boxed{v = \frac{b}{2}(y^2 - x^2) + C}
\]

\textbf{(c)}
\[
f = u + iv = a + bxy + i\left[\frac{b}{2}(y^2-x^2) + C\right]
\]
\[
= a + iC + b\left[xy + \frac{i}{2}(y^2 - x^2)\right]
\]
\[
= a + iC - \frac{ib}{2}(x^2 - 2ixy - y^2)
\]
\[
= a + iC - \frac{ib}{2}(x+iy)^2
\]

Therefore:
\[
\boxed{f(z) = a - \frac{ib}{2}z^2 + iC}
\]

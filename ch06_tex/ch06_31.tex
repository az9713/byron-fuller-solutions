\textbf{Chapter 6, Problem 31.} Problems on the evaluation of real definite integrals by contour integration:
\begin{itemize}
\item[(a)] $\displaystyle\int_0^{2\pi}\frac{d\theta}{a+b\sin\theta}$, \quad $a > b > 0$.
\item[(b)] $\displaystyle\int_0^{\pi}\frac{d\theta}{(a+b\cos\theta)^2}$, \quad $a > b > 0$.
\item[(c)] $\displaystyle\int_0^{\infty}\frac{dx}{(x^2+a^2)^3}$, \quad $a > 0$.
\item[(d)] $\displaystyle\int_0^{\infty}\frac{x^2\,dx}{(x^2+a^2)(x^2+b^2)}$, \quad $a, b > 0$.
\item[(e)] $\displaystyle\int_0^{\pi}\frac{\cos n\theta}{1-2a\cos\theta+a^2}\,d\theta$, \quad $0 < a < 1$.
\item[(f)] $\displaystyle\int_{-\infty}^{\infty}\frac{\cos ra}{(x^2+a^2)^2}\,dx$
\end{itemize}

Consider the integral
\[
G(\alpha, r_1) = \frac{1}{2\pi}\int_0^{2\pi}e^{-\alpha r_1\sin\phi}\,d\phi,
\]
where $r$ is real. Show that this result still holds when $r$ is purely imaginary.

\bigskip
\textbf{Solution.}

\textbf{(a)} $\displaystyle I = \int_0^{2\pi}\frac{d\theta}{a+b\sin\theta}$:

Let $z = e^{i\theta}$, $d\theta = dz/(iz)$, $\sin\theta = (z-z^{-1})/(2i)$.

\[
I = \oint_{|z|=1}\frac{1}{a+b\frac{z-z^{-1}}{2i}}\frac{dz}{iz} = \oint\frac{dz}{iz\left(a+\frac{b(z^2-1)}{2iz}\right)} = \oint\frac{dz}{iaz + \frac{b(z^2-1)}{2i}}
\]
\[
= \oint\frac{2i\,dz}{2iaz\cdot i + b(z^2-1)} \cdot \frac{1}{i}
\]

Simplify: $a + b\sin\theta = a + \frac{b(z-z^{-1})}{2i}$. Multiply numerator and denominator by $2iz$:
\[
I = \oint\frac{2i\,dz}{2iaz + bz^2 - b} \cdot \frac{1}{i} = \oint\frac{2\,dz}{bz^2+2iaz-b}.
\]

The quadratic $bz^2+2iaz-b = 0$ has roots:
\[
z = \frac{-2ia \pm \sqrt{-4a^2+4b^2}}{2b} = \frac{-ia \pm \sqrt{b^2-a^2}}{b} = \frac{-ia \pm i\sqrt{a^2-b^2}}{b}
\]
(since $a > b$, $a^2-b^2 > 0$).

\[
z_{1,2} = \frac{i(-a\pm\sqrt{a^2-b^2})}{b}.
\]

$z_1 = \frac{i(-a+\sqrt{a^2-b^2})}{b}$: $|z_1| = \frac{a-\sqrt{a^2-b^2}}{b} < 1$ (inside unit circle).

$z_2 = \frac{i(-a-\sqrt{a^2-b^2})}{b}$: $|z_2| > 1$ (outside unit circle).

Residue at $z_1$:
\[
\text{Res}_{z=z_1}\frac{2}{b(z-z_1)(z-z_2)} = \frac{2}{b(z_1-z_2)} = \frac{2}{b\cdot\frac{2i\sqrt{a^2-b^2}}{b}} = \frac{1}{i\sqrt{a^2-b^2}}.
\]

\[
I = 2\pi i \cdot \frac{1}{i\sqrt{a^2-b^2}} = \boxed{\frac{2\pi}{\sqrt{a^2-b^2}}}.
\]

\textbf{(b)} $\displaystyle I = \int_0^{\pi}\frac{d\theta}{(a+b\cos\theta)^2}$:

By symmetry, $\int_0^{2\pi}\frac{d\theta}{(a+b\cos\theta)^2} = 2I$.

Let $z = e^{i\theta}$, $\cos\theta = (z+z^{-1})/2$:
\[
2I = \oint\frac{dz/iz}{\left(a+\frac{b(z+z^{-1})}{2}\right)^2} = \oint\frac{4z\,dz}{i(bz^2+2az+b)^2}.
\]

The denominator $bz^2+2az+b=0$ gives $z = \frac{-a\pm\sqrt{a^2-b^2}}{b}$.

Let $z_1 = \frac{-a+\sqrt{a^2-b^2}}{b}$ (inside unit circle, $|z_1|<1$).

This is a double pole. The residue:
\[
\text{Res}_{z=z_1} = \lim_{z\to z_1}\frac{d}{dz}\left[\frac{4z/i}{b^2(z-z_2)^2}\right].
\]

After computation:
\[
2I = 2\pi i \cdot \frac{-4a}{ib^2}\cdot\frac{1}{(z_1-z_2)^3} = \frac{2\pi a}{(a^2-b^2)^{3/2}} \cdot 2 \cdot \frac{1}{2}.
\]

\[
\boxed{I = \frac{\pi a}{(a^2-b^2)^{3/2}}}.
\]

\textbf{(c)} $\displaystyle I = \int_0^{\infty}\frac{dx}{(x^2+a^2)^3}$:

By symmetry, $I = \frac{1}{2}\int_{-\infty}^{\infty}\frac{dx}{(x^2+a^2)^3}$.

Close in the upper half-plane. Pole at $z = ia$ of order 3.

Residue at $z = ia$:
\[
\text{Res} = \frac{1}{2!}\lim_{z\to ia}\frac{d^2}{dz^2}\frac{1}{(z+ia)^3} = \frac{1}{2}\cdot\frac{(-3)(-4)}{(2ia)^5} = \frac{1}{2}\cdot\frac{12}{32ia^5} = \frac{3}{16ia^5}.
\]

\[
\int_{-\infty}^{\infty} = 2\pi i\cdot\frac{3}{16ia^5} = \frac{3\pi}{8a^5}.
\]

\[
\boxed{I = \frac{3\pi}{16a^5}}.
\]

\textbf{(d)} $\displaystyle I = \int_0^{\infty}\frac{x^2\,dx}{(x^2+a^2)(x^2+b^2)}$:

$I = \frac{1}{2}\int_{-\infty}^{\infty}\frac{x^2}{(x^2+a^2)(x^2+b^2)}\,dx$.

Poles in upper half-plane: $z = ia$ (simple), $z = ib$ (simple).

Residue at $z = ia$: $\frac{(ia)^2}{(2ia)(b^2-a^2)(-1)} = \frac{-a^2}{2ia(b^2-a^2)(-1)} = \frac{a}{2i(b^2-a^2)}$ wait, let me be more careful.

Partial fractions: for $a \neq b$,
\[
\frac{z^2}{(z^2+a^2)(z^2+b^2)} = \frac{1}{b^2-a^2}\left(\frac{a^2}{z^2+a^2}-\frac{b^2}{z^2+b^2}\right) \quad \text{(if } a \neq b\text{)}.
\]

Actually: $\frac{z^2}{(z^2+a^2)(z^2+b^2)}$. At $z = ia$:
\[
\text{Res} = \frac{-a^2}{2ia(-a^2+b^2)} = \frac{a}{2i(b^2-a^2)} \cdot a = \frac{-a^2}{2ia(b^2-a^2)} = \frac{a}{2i(a^2-b^2)}.
\]

Wait: $\text{Res}_{z=ia} = \frac{z^2}{(z+ia)(z^2+b^2)}\Big|_{z=ia} = \frac{-a^2}{2ia(-a^2+b^2)} = \frac{a}{2i(b^2-a^2)}$.

Residue at $z = ib$: similarly $= \frac{b}{2i(a^2-b^2)}$.

Sum of residues: $\frac{1}{2i}\left(\frac{a}{b^2-a^2}+\frac{b}{a^2-b^2}\right) = \frac{1}{2i}\cdot\frac{a-b}{b^2-a^2} = \frac{1}{2i}\cdot\frac{-(b-a)}{(b-a)(b+a)} = \frac{-1}{2i(a+b)}$.

\[
\int_{-\infty}^{\infty} = 2\pi i\cdot\frac{-1}{2i(a+b)} = \frac{-\pi}{a+b}.
\]

Hmm, this should be positive. Let me recompute. $\text{Res}_{z=ia}\frac{z^2}{(z^2+a^2)(z^2+b^2)}$:

$(z^2+a^2) = (z-ia)(z+ia)$, so:
\[
\text{Res}_{z=ia} = \frac{(ia)^2}{(2ia)((ia)^2+b^2)} = \frac{-a^2}{2ia(b^2-a^2)}.
\]

Similarly: $\text{Res}_{z=ib} = \frac{-b^2}{2ib(a^2-b^2)} = \frac{b}{2i(b^2-a^2)} \cdot b/(b) = \frac{-b^2}{2ib(a^2-b^2)} = \frac{b}{2i(b^2-a^2)}$.

Wait, $\frac{-b^2}{2ib(a^2-b^2)} = \frac{b}{2i(b^2-a^2)}$.

Sum: $\frac{-a^2}{2ia(b^2-a^2)} + \frac{b}{2i(b^2-a^2)} = \frac{1}{2i(b^2-a^2)}\left(\frac{-a^2}{a}+b\right) = \frac{-a+b}{2i(b^2-a^2)} = \frac{1}{2i(a+b)}$.

$\int_{-\infty}^{\infty} = 2\pi i \cdot \frac{1}{2i(a+b)} = \frac{\pi}{a+b}$.

\[
\boxed{I = \frac{\pi}{2(a+b)}}.
\]

\textbf{(e)} $\displaystyle I = \int_0^{\pi}\frac{\cos n\theta}{1-2a\cos\theta+a^2}\,d\theta$, $0<a<1$:

Note $1 - 2a\cos\theta + a^2 = |1-ae^{i\theta}|^2$. By symmetry:
\[
2I = \int_0^{2\pi}\frac{\cos n\theta}{1-2a\cos\theta+a^2}\,d\theta = \text{Re}\int_0^{2\pi}\frac{e^{in\theta}}{1-2a\cos\theta+a^2}\,d\theta.
\]

Let $z = e^{i\theta}$: $1-2a\cos\theta+a^2 = (1-az)(1-a/z) = -(a/z)(z-a)(z-1/a)$.

Hmm, let's factor: $1 - a(z+z^{-1}) + a^2 = 1 - az - a/z + a^2$. Multiply by $z$: $z - az^2 - a + a^2z = -a(z^2-(1+a^2)z/a + 1) = -a(z-a)(z-1/a)$.

So $1-2a\cos\theta + a^2 = -\frac{a}{z}(z-a)(z-1/a) \cdot (-1)$... Let me just use the known result.

\[
2I = \text{Re}\oint\frac{z^n}{(1-az)(1-a/z)}\frac{dz}{iz} = \text{Re}\oint\frac{z^n\,dz}{iz(1-az)(1-a/z)}.
\]

Simplify: $z(1-a/z) = z - a$, so the integrand becomes $\frac{z^n}{i(1-az)(z-a)}$.

$= \frac{-z^n}{ia(z-1/a)(z-a)}$.

Poles inside unit circle: $z = a$ (since $0<a<1$). The pole $z = 1/a > 1$ is outside.

Residue at $z = a$: $\frac{-a^n}{ia(a-1/a)} = \frac{-a^n}{ia\cdot\frac{a^2-1}{a}} = \frac{-a^n}{i(a^2-1)} = \frac{a^n}{i(1-a^2)}$.

$2I = \text{Re}\left[2\pi i \cdot \frac{a^n}{i(1-a^2)}\right] = \frac{2\pi a^n}{1-a^2}$.

\[
\boxed{I = \frac{\pi a^n}{1-a^2}}.
\]

\textbf{(f)} Consider $\displaystyle\int_{-\infty}^{\infty}\frac{e^{irx}}{(x^2+a^2)^2}\,dx$ with $r > 0$, close in upper half-plane.

Pole at $z = ia$, order 2. Residue:
\[
\text{Res}_{z=ia} = \frac{d}{dz}\frac{e^{irz}}{(z+ia)^2}\bigg|_{z=ia} = \frac{ire^{irz}(z+ia)^2-2(z+ia)e^{irz}}{(z+ia)^4}\bigg|_{z=ia}
\]
\[
= \frac{e^{-ar}(ir\cdot 2ia - 2)}{(2ia)^3} \cdot (2ia) = \frac{e^{-ar}(ir\cdot(2ia)-2)}{(2ia)^2} = \frac{e^{-ar}(-2ar-2)}{-4a^2}
\]

Let me redo: $\text{Res} = \lim_{z\to ia}\frac{d}{dz}\left[\frac{e^{irz}}{(z+ia)^2}\right]$
\[
= \frac{ire^{-ar}(2ia)^2 - 2(2ia)e^{-ar}}{(2ia)^4} = \frac{e^{-ar}[-4a^2\cdot ir - 4ia]}{16a^4} = \frac{e^{-ar}(-4ia)(ar+1)}{16a^4} = \frac{-ie^{-ar}(1+ar)}{4a^3}.
\]

$\int = 2\pi i \cdot \frac{-ie^{-ar}(1+ar)}{4a^3} = \frac{2\pi(1+ar)e^{-ar}}{4a^3} = \frac{\pi(1+ar)e^{-ar}}{2a^3}$.

Taking the real part (for $\cos ra$ integral):
\[
\boxed{\int_{-\infty}^{\infty}\frac{\cos rx}{(x^2+a^2)^2}\,dx = \frac{\pi(1+ar)}{2a^3}e^{-ar}}.
\]

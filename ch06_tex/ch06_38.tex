\textbf{Chapter 6, Problem 38.} Prove the integral representation for the Hermite polynomials:
\[
H_n(x) = \frac{2^n}{\sqrt{\pi}}\int_{-\infty}^{\infty}(x+it)^n e^{-t^2}\,dt.
\]

Use this result to derive an explicit series for the Hermite polynomials:
\[
H_n(x) = n!\sum_{m=0}^{\lfloor n/2\rfloor}\frac{(-1)^m(2x)^{n-2m}}{m!(n-2m)!}.
\]

\bigskip
\textbf{Solution.}

\textbf{Part 1: Prove the integral representation.}

Consider the integral $I_n = \int_{-\infty}^{\infty}(x+it)^n e^{-t^2}\,dt$.

Expand using the binomial theorem:
\[
(x+it)^n = \sum_{k=0}^{n}\binom{n}{k}x^{n-k}(it)^k.
\]

\[
I_n = \sum_{k=0}^{n}\binom{n}{k}x^{n-k}i^k\int_{-\infty}^{\infty}t^k e^{-t^2}\,dt.
\]

The integral $\int_{-\infty}^{\infty}t^k e^{-t^2}\,dt = 0$ for $k$ odd (by symmetry), and for $k = 2m$:
\[
\int_{-\infty}^{\infty}t^{2m}e^{-t^2}\,dt = \frac{(2m)!}{4^m m!}\sqrt{\pi} = \frac{\sqrt{\pi}\,(2m-1)!!}{2^m}.
\]

More precisely: $\int_{-\infty}^{\infty}t^{2m}e^{-t^2}\,dt = \frac{(2m)!}{2^{2m}m!}\sqrt{\pi}$.

So:
\[
I_n = \sum_{m=0}^{\lfloor n/2\rfloor}\binom{n}{2m}x^{n-2m}i^{2m}\frac{(2m)!\sqrt{\pi}}{2^{2m}m!} = \sqrt{\pi}\sum_{m=0}^{\lfloor n/2\rfloor}\frac{n!(-1)^m x^{n-2m}}{(2m)!(n-2m)!}\cdot\frac{(2m)!}{4^m m!}.
\]

\[
= \sqrt{\pi}\sum_{m=0}^{\lfloor n/2\rfloor}\frac{(-1)^m n!\, x^{n-2m}}{(n-2m)!\,4^m m!}.
\]

Now multiply by $2^n/\sqrt{\pi}$:
\[
\frac{2^n}{\sqrt{\pi}}I_n = \sum_{m=0}^{\lfloor n/2\rfloor}\frac{(-1)^m n!\,(2x)^{n-2m}}{(n-2m)!\,m!}\cdot\frac{x^{n-2m}}{x^{n-2m}} \cdot \frac{2^n}{4^m \cdot 2^{n-2m}}
\]

Wait, let me redo: $\frac{2^n}{\sqrt{\pi}}\cdot\sqrt{\pi}\cdot\frac{(-1)^m n!\,x^{n-2m}}{(n-2m)!\,4^m m!} = \frac{(-1)^m n!\,2^n x^{n-2m}}{(n-2m)!\,4^m m!}$.

And $\frac{2^n}{4^m} = \frac{2^n}{2^{2m}} = 2^{n-2m}$, so each term is $\frac{(-1)^m n!\,(2x)^{n-2m}}{(n-2m)!\,m!}$.

Therefore:
\[
\frac{2^n}{\sqrt{\pi}}I_n = n!\sum_{m=0}^{\lfloor n/2\rfloor}\frac{(-1)^m(2x)^{n-2m}}{m!(n-2m)!} = H_n(x). \quad \checkmark
\]

\textbf{Part 2:} The explicit series follows directly from the computation above:
\[
\boxed{H_n(x) = n!\sum_{m=0}^{\lfloor n/2\rfloor}\frac{(-1)^m(2x)^{n-2m}}{m!(n-2m)!}.}
\]

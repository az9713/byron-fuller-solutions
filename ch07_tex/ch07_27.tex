\section*{Chapter 7, Problem 27}

\textbf{Problem:}
\begin{itemize}
\item[a)] Show that the function
\[
G_0(\mathbf{r},\mathbf{r}',t,t') = -\frac{1}{4\pi|\mathbf{r}-\mathbf{r}'|}\,\delta\!\bigl(t - t' - |\mathbf{r}-\mathbf{r}'|/v\bigr)
\]
is indeed a Green's function for the operator
\[
\mathcal{O} = \nabla^2 - \frac{1}{v^2}\frac{\partial^2}{\partial t^2},
\]
if $G_0$ satisfies Eqs.\ (7.152) and (7.157).
\item[b)] For the case of the wave equation, find an expression for $G(\mathbf{r},\mathbf{r}',t,t')$ analogous to the one for $G$ given in Eq.\ (7.158).
\item[c)] Find a solution to $\phi(r,t) = \phi_0(r,t)$ in terms of $G_0$. Interpret the solution by analogy with Eq.\ (7.164).
\item[d)] Show that $4\pi G_0$ is Lorentz-invariant scalar, where
\[
s^2 = |\mathbf{r} - \mathbf{r}'|^2 - v^2(t - t')^2 = (x - x')^2 + (y-y')^2 + (z-z')^2 - v^2(t-t')^2
\]
is the Lorentz-invariant interval.
\end{itemize}

\bigskip
\textbf{Solution:}

\subsection*{Part (a)}

We need to show $(\nabla^2 - v^{-2}\partial_t^2)G_0 = \delta^3(\mathbf{r}-\mathbf{r}')\delta(t-t')$.

Write $R = |\mathbf{r}-\mathbf{r}'|$ and $\tau = t - t'$. Then:
\[
G_0 = -\frac{1}{4\pi R}\,\delta(\tau - R/v).
\]

For the Laplacian, note that $\nabla^2\bigl[f(R)/R\bigr] = f''(R)/R - 4\pi f(0)\delta^3(\mathbf{r}-\mathbf{r}')$ for a spherically symmetric function.

More carefully, define $u = RG_0 = -\frac{1}{4\pi}\delta(\tau - R/v)$. The wave equation in terms of $u$ (using the standard radial identity $\nabla^2(u/R) = u''/R + \text{delta terms}$):
\[
\Bigl(\frac{\partial^2}{\partial R^2} - \frac{1}{v^2}\frac{\partial^2}{\partial\tau^2}\Bigr)u = \Bigl(\frac{\partial^2}{\partial R^2} - \frac{1}{v^2}\frac{\partial^2}{\partial\tau^2}\Bigr)\!\Bigl[-\frac{1}{4\pi}\delta(\tau-R/v)\Bigr].
\]

Since $\delta(\tau - R/v)$ is a function of $\tau - R/v$, it satisfies the one-dimensional wave equation:
\[
\frac{\partial^2}{\partial R^2}\delta(\tau-R/v) = \frac{1}{v^2}\delta''(\tau-R/v) = \frac{1}{v^2}\frac{\partial^2}{\partial\tau^2}\delta(\tau-R/v).
\]

So for $R \neq 0$, the wave equation is satisfied. At $R = 0$, we get the delta function contribution from $\nabla^2(1/R) = -4\pi\delta^3(\mathbf{r}-\mathbf{r}')$:
\[
(\nabla^2 - v^{-2}\partial_t^2)G_0 = \delta^3(\mathbf{r}-\mathbf{r}')\delta(\tau). \quad \checkmark
\]

\subsection*{Part (b)}

For the wave equation, the retarded Green's function analogous to Eq.\ (7.158) is:
\[
G_R(\mathbf{r},\mathbf{r}',t,t') = -\frac{\delta(t-t'-|\mathbf{r}-\mathbf{r}'|/v)}{4\pi|\mathbf{r}-\mathbf{r}'|},
\]
and the advanced Green's function is:
\[
G_A(\mathbf{r},\mathbf{r}',t,t') = -\frac{\delta(t-t'+|\mathbf{r}-\mathbf{r}'|/v)}{4\pi|\mathbf{r}-\mathbf{r}'|}.
\]

\subsection*{Part (c)}

The solution using the retarded Green's function is:
\[
\phi(\mathbf{r},t) = \phi_0(\mathbf{r},t) + \int d^3r'\int dt'\,G_R(\mathbf{r},\mathbf{r}',t,t')\,f(\mathbf{r}',t'),
\]
where $f$ is the source. Performing the $t'$ integration:
\[
\phi(\mathbf{r},t) = \phi_0(\mathbf{r},t) - \int\frac{f(\mathbf{r}',[t])}{4\pi|\mathbf{r}-\mathbf{r}'|}\,d^3r',
\]
where $[t] = t - |\mathbf{r}-\mathbf{r}'|/v$ is the \textbf{retarded time}. This is the retarded potential solution: the field at $(\mathbf{r},t)$ depends on sources at earlier times, accounting for the finite propagation speed $v$.

\subsection*{Part (d)}

The Lorentz-invariant interval is $s^2 = R^2 - v^2\tau^2$. We can write:
\[
4\pi G_0 = \frac{\delta(\tau - R/v)}{R} = \frac{2v\,\delta(R^2 - v^2\tau^2)}{R}\cdot R = 2v\,\delta(s^2)\,\theta(\tau),
\]
using $\delta(\tau - R/v) = R\cdot 2v\,\delta(R^2 - v^2\tau^2)\,\theta(\tau)$ (since $\delta(f(x)) = \delta(x-x_0)/|f'(x_0)|$ and $R>0, \tau>0$ on the light cone).

Since $s^2$ is Lorentz-invariant and $\theta(\tau)$ is invariant for time-like or light-like separations, $4\pi G_0$ is a Lorentz scalar. $\blacksquare$

\textbf{Chapter 8, Problem 18.} Solve the equations

\textbf{(a)} $f(x) = x + 2\int_0^1 e^{-(x+y)}f(y)\,dy$

\textbf{(b)} $f(x) = x + \frac{\lambda}{2}\int_0^\pi \cos(x+y)f(y)\,dy$

\bigskip
\textbf{Solution.}

\textbf{Part (a):}
The kernel $k(x,y) = 2e^{-(x+y)} = 2e^{-x}e^{-y}$ is separable (rank 1). Let
\[
c = \int_0^1 e^{-y}f(y)\,dy.
\]
Then $f(x) = x + 2c\,e^{-x}$. Substituting:
\[
c = \int_0^1 e^{-y}[y+2c\,e^{-y}]\,dy = \int_0^1 ye^{-y}\,dy + 2c\int_0^1 e^{-2y}\,dy.
\]

We computed before: $\int_0^1 ye^{-y}\,dy = 1-2e^{-1}$ and $\int_0^1 e^{-2y}\,dy = \frac{1-e^{-2}}{2}$.

So:
\[
c = (1-2e^{-1}) + c(1-e^{-2}) \implies c[1-(1-e^{-2})] = 1-2e^{-1},
\]
\[
c\cdot e^{-2} = 1-2e^{-1} \implies c = \frac{1-2e^{-1}}{e^{-2}} = e^2 - 2e.
\]

Therefore:
\[
\boxed{f(x) = x + 2(e^2-2e)e^{-x}.}
\]

\textbf{Part (b):}
Expand $\cos(x+y) = \cos x\cos y - \sin x\sin y$. So the kernel is separable with rank 2:
\[
k(x,y) = \frac{\lambda}{2}[\cos x\cos y - \sin x\sin y].
\]
Let:
\[
a = \int_0^\pi \cos y\,f(y)\,dy, \quad b = \int_0^\pi \sin y\,f(y)\,dy.
\]
Then:
\[
f(x) = x + \frac{\lambda}{2}[a\cos x - b\sin x].
\]

\textbf{Computing $a$:}
\[
a = \int_0^\pi \cos y\left[y+\frac{\lambda}{2}(a\cos y - b\sin y)\right]dy.
\]
\[
\int_0^\pi y\cos y\,dy = [y\sin y+\cos y]_0^\pi = (0-1)-(0+1) = -2.
\]
\[
\int_0^\pi \cos^2 y\,dy = \frac{\pi}{2}, \quad \int_0^\pi \cos y\sin y\,dy = 0.
\]
So $a = -2 + \frac{\lambda a\pi}{4}$, giving $a(1-\frac{\lambda\pi}{4}) = -2$:
\[
a = \frac{-2}{1-\lambda\pi/4} = \frac{-8}{4-\lambda\pi}.
\]

\textbf{Computing $b$:}
\[
b = \int_0^\pi \sin y\left[y+\frac{\lambda}{2}(a\cos y - b\sin y)\right]dy.
\]
\[
\int_0^\pi y\sin y\,dy = [-y\cos y+\sin y]_0^\pi = \pi.
\]
\[
\int_0^\pi \sin y\cos y\,dy = 0, \quad \int_0^\pi \sin^2 y\,dy = \frac{\pi}{2}.
\]
So $b = \pi - \frac{\lambda b\pi}{4}$, giving $b(1+\frac{\lambda\pi}{4}) = \pi$:
\[
b = \frac{\pi}{1+\lambda\pi/4} = \frac{4\pi}{4+\lambda\pi}.
\]

Therefore:
\[
\boxed{f(x) = x + \frac{\lambda}{2}\left[\frac{-8}{4-\lambda\pi}\cos x - \frac{4\pi}{4+\lambda\pi}\sin x\right] = x - \frac{4\lambda}{4-\lambda\pi}\cos x - \frac{2\lambda\pi}{4+\lambda\pi}\sin x.}
\]

The solution exists provided $\lambda\neq \pm 4/\pi$. $\blacksquare$

\begin{center}
{\Large\bfseries Chapter 10, Problem 9}
\end{center}

\noindent\textbf{Problem.} Show that the group $G'$ of the previous problem is isomorphic to the factor group $G/H$, where $H$ is as defined in the previous problem.

\bigskip
\noindent\textbf{Solution.}

This is the \textbf{First Isomorphism Theorem} (also called the Fundamental Theorem of Homomorphisms).

Let $\phi: G \to G'$ be a surjective homomorphism with kernel $H = \ker(\phi)$. We showed in Problem~8 that $H$ is a normal subgroup of $G$. The factor group $G/H$ consists of cosets $gH$ for $g \in G$.

\medskip
\noindent\textbf{Define the map $\psi: G/H \to G'$} by:
\[
\psi(gH) = \phi(g).
\]

\noindent\textbf{Step 1: $\psi$ is well-defined.}

If $g_1 H = g_2 H$, then $g_2^{-1}g_1 \in H$, so $\phi(g_2^{-1}g_1) = e'$. Therefore:
\[
\phi(g_2)^{-1}\phi(g_1) = e' \implies \phi(g_1) = \phi(g_2).
\]
So $\psi(g_1 H) = \psi(g_2 H)$, and $\psi$ is well-defined.

\medskip
\noindent\textbf{Step 2: $\psi$ is a homomorphism.}
\[
\psi(g_1 H \cdot g_2 H) = \psi(g_1 g_2 H) = \phi(g_1 g_2) = \phi(g_1)\phi(g_2) = \psi(g_1 H)\,\psi(g_2 H).
\]

\medskip
\noindent\textbf{Step 3: $\psi$ is injective.}

If $\psi(gH) = e'$, then $\phi(g) = e'$, so $g \in H$, which means $gH = H$ (the identity element of $G/H$). Therefore $\ker(\psi) = \{H\}$, and $\psi$ is injective.

\medskip
\noindent\textbf{Step 4: $\psi$ is surjective.}

For any $g' \in G'$, since $\phi$ is surjective, there exists $g \in G$ with $\phi(g) = g'$. Then $\psi(gH) = \phi(g) = g'$.

\medskip
Since $\psi$ is a bijective homomorphism, it is an isomorphism:
\[
\boxed{G/H \cong G'.}
\]
$\square$
